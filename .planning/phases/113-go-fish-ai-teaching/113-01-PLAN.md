export interface BotMove {
  action: string;                  // e.g. 'ask'
  args: Record<string, unknown>;   // go-fish ask: { target: number, rank: string }
}

Hint-target hook signature (BoardSmith game-session #extractMoveTarget):
  (move: BotMove) => ElementRef | undefined
  ElementRef = { id?: number; name?: string; notation?: string; className?: string }
  Receives the move ONLY (no game/seat) — a rank-keyed { name } is the correct ref; a specific card id is NOT derivable here.

Checkers analog (proven) — checkers/src/rules/index.ts:
```typescript
ai: {
  objectives: getCheckersObjectives,
  hintTargetFromMove: (move) => {
    const dest = move.args.destination as { toNotation?: string } | undefined;
    return dest?.toNotation ? { notation: dest.toNotation } : undefined;
  },
},
```

go-fish ask move (src/rules/actions.ts createAskAction): BotMove { action:'ask', args:{ target:number(seat), rank:string } }. Rank values are bare ranks ('2'..'10','J','Q','K','A'). Individual card names are rank+suit ('7H').
